一.初始化串口Init_UART

1.打开串口所对应的GPIO时钟

RCC_APB2PeriphResetCmd(RCC_APB2Periph_GPIOA, ENABLE);

2.打开串口所对应时钟

RCC_APB2PeriphResetCmd(RCC_APB2Periph_USART1, ENABLE);

3.设置GPIO模式（结构体）GPIO_INIT

TX推挽输出PA9

GPIO_InitStruct.GPIO_Mode = GPIO_Mode_AF_PP;
GPIO_InitStruct.GPIO_Pin = GPIO_Pin_9;
GPIO_InitStruct.GPIO_Speed = GPIO_Speed_50MHz;
GPIO_Init(GPIOA, &GPIO_InitStruct);

RX浮空输入PA10

GPIO_InitStruct.GPIO_Mode = GPIO_Mode_IN_FLOATING;
GPIO_InitStruct.GPIO_Pin = GPIO_Pin_10;
GPIO_Init(GPIOA, &GPIO_InitStruct);

4.设置串口模式（结构体）USART_INIT

USART_InitStruct.USART_BaudRate = 115200;
USART_InitStruct.USART_HardwareFlowControl = USART_HardwareFlowControl_None;
USART_InitStruct.USART_Mode = USART_Mode_Rx | USART_Mode_Tx;
USART_InitStruct.USART_Parity = USART_Parity_No;
USART_InitStruct.USART_StopBits = USART_StopBits_1;
USART_InitStruct.USART_WordLength = USART_WordLength_8b;
USART_Init(USART1, &USART_InitStruct);

5.开启串口usart_cmd()

NVIC_Config_UART();//总中断
USART_ITConfig(USART1,USART_IT_RXNE,ENABLE);//使能接收中断
若是有接收中断则开启串口前需要打开总中断和使能接收中断
USART_Cmd(USART1,ENABLE);
  • 串口发送不需要中断，串口接收需要接收中断
    1.中断接收
    2.getchar/scanf接收

二.串口发送uart_sentbyte()

void UART_SendByte(USART_TypeDef * USARTx,uint8_t data)
{
    USART_SendData(USARTx,  data);
    while(!USART_GetFlagStatus(USARTx, USART_FLAG_TXE));    

}

1.调用usart_senddata(USARTx,data)

2.检验串口是否发送完毕

USART_GetFlagStatus()//检验串口发送寄存器是否为空，为空flag置1

三.发送字符串uart_sentstring()

void UART_SendString(USART_TypeDef * USARTx,char* string)

{
    char *str = string;
    while(*str)
    {
        UART_SendByte(USARTx, *str);
        str++;    
    }
}

1.就是做一只调用前面写的uart_sentbyte()

下面是介绍调用char *string 的用发，便于理解

1、char *string=“hello world!”;
（1）给变量baistring分配一个地址用来存放dustring的值，注意其zhi值是一个指dao向字符的指针。
（2）在常量区建立一个字符串“hello world!”。
（3）把字符串首字符h的地址赋值给string变量。

2、 用char string[ ]=“hello world”;

（1）系统为数组string分配空间。
（2）在为string分配的空间依次存放字符串“hello world”; 一直放完为止。并在最后放一个0。
注意：第一种情况不可把string理解为数组，第二种情况不可理解string是指针。二者很大不同，切切！！，

四.中断接收 NVIC_ConFig_UART()

  • 开启串口前调用，相当于51里面的总中断

  • 配置NVIC

void NVIC_Config_UART(void)
{
    NVIC_InitTypeDef NVIC_InitStrct;
    NVIC_InitStrct.NVIC_IRQChannel = USART1_IRQn;
    NVIC_InitStrct.NVIC_IRQChannelCmd = ENABLE;
    NVIC_InitStrct.NVIC_IRQChannelPreemptionPriority = 1;
    NVIC_InitStrct.NVIC_IRQChannelSubPriority = 1;
    NVIC_PriorityGroupConfig(NVIC_PriorityGroup_2);    

    NVIC_Init(&NVIC_InitStrct);
}

中断由NVIC管理
1.中断优先级的分组
NVIC_PriorityGroupConfig(NVIC_PriorityGroup_2);

  • 两个主优先级，两个子优先级
  • NVIC配置优先级有个寄存器有四位，四位存放的是主优先级和子优先级

2.NVIC_Init()

参数是结构体
uint8_t NVIC_IRQChannel;//谁的中断
uint8_t NVIC_IRQChannelPreemptionPriority;//首要优先级,抢占优先级
uint8_t NVIC_IRQChannelSubPriority;//子优先级
FunctionalState NVIC_IRQChannelCmd;//启用还是禁用

五.先打开总中断，再打开串口中断USART_ITConfig();

NVIC_Config_UART();//总中断
USART_ITConfig(USART1,USART_IT_RXNE,ENABLE);//使能接收中断
//若是有接收中断则开启串口前需要打开总中断和使能接收中断

这个在前面的开启串口usart_cmd()已经介绍了。

六.写中断处理函数

  • 写在stm32f10x_it.c里面，interrupt

  • 串口中断中断函数名字是固定的不能更改

void USART1_IRQHandler(void)
{
    uint8_t temp;
    if(USART_GetFlagStatus(USART1,USART_IT_RXNE))
    {
        temp = USART_ReceiveData(USART1);
        USART_SendData(USART1,temp);    
    }

}
